# 创建命令空间
kubectl create ns dev 
# 同上一模一样的效果
kubectl create namespace dev
# 删除一个资源
#namespaces 表示一种资源类型
kubectl delete namespaces dev
kubectl delete ns/dev #同上, 可以指定多个 
kubectl delete ns/dev ns/test

# 这个是陈述式的 语句
# 如果已经创建过,则会报错
kubectl create -f dev-ns.yaml
kubectl create -f dashboard.yaml
# 这个是申明式的语句,
#这个不会报错,重新应用一遍
kubectl apply -f dashboard.yaml #(推荐) 
# 查看状态
kubectl get -f myapp.yaml
#可以指定一个目录,一次创建多个
kubectl apply -f ./

# 注意这个 会先删除pod 再重建, 会导致Pod的IP地址和标识符uid发生变化
kubectl replace --force -f dashboard.yaml
k get po -o yaml # 看到的uid不一样了,/var/lib/kubelet/pods/

# 创建一个deploy
kubectl create deploy ngx-dep  --image=nginx:1.14-alpine

6.2 使用yaml文件创建

kubectl run nginx-test --image=nginx  --dry-run -oyaml  >nginx.yaml

    #--dry-run is deprecated and can be replaced with --dry-run=client.
    apiVersion: v1
    kind: Pod #资源类型
    metadata: #元数据
      creationTimestamp: null
      labels: # 标签
        run: nginx-test # key value
      name: nginx-test #pod 名称 
    spec: #pod 真正的配置
      containers: #配置容器的地方,可以定义多个
      - image: nginx
        name: nginx-test
        ports:
        - containerPort: 80
        args: ["abc"] #相当于docker里的CMD
        command: [ "sleep" ,"10" ] # 相当于docker里的entrypoint, 可以修改容器的启动命令
        
     #  - image: redis
     #   name: redis
        resources: {}
      dnsPolicy: ClusterFirst
      restartPolicy: Always
    status: {}

# 如果yaml文件里有多个配置(---隔开的). 可以选择创建哪一个.
k create -f sidecar.yaml  -l run=nginx-test

6.5 查看pod的日志

# 容器里日志打在前台的可以这样直接看
k logs nginx-test
# 最近10m中内的日志 , --since=1h 一个小时
k logs  nginx --since=10m
# 实时查看
k logs -f nginx-test
# 查看最新的100行
k logs --tail=100 nginx-test
# Pod之前发生过崩溃,访问上一个Pod的日志
k logs --previous nginx-test
# k logs pod名 容器名
k logs pod-nginx c-nginx1 -n test
# 如果容器里日志写在它里面的某个文件, 那就 exec 进去容器 查看
k exec nginx-test -- tail -f /var/logs/xxxxx
